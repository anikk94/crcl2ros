142 inline tf::Matrix3x3 Rpy2RotMatrix(double r, double p, double y)
143 {
144  double sa, sb, sg;
145  double ca, cb, cg;
146  tf::Matrix3x3 m;
147 
148  sa = sin(y);
149  sb = sin(p);
150  sg = sin(r);
151 
152 
153  ca = cos(y);
154  cb = cos(p);
155  cg = cos(r);
156 
157 
158  // m is row/column
159  m[0][0] = ca * cb;
160  m[1][0] = ca * sb * sg - sa * cg;
161  m[2][0] = ca * sb * cg + sa * sg;
162 
163 
164  m[0][1] = sa * cb;
165  m[1][1] = sa * sb * sg + ca * cg;
166  m[2][1] = sa * sb * cg - ca * sg;
167 
168 
169  m[0][2] = -sb;
170  m[1][2] = cb * sg;
171  m[2][2] = cb * cg;
172 
173 
174  return m;
175 }

187 inline int RotMatrix2Rpy(tf::Matrix3x3 m, double &r, double &p, double &y)
188 {
189  p = atan2(-m[0][2], sqrt(SQ(m[0][0]) + SQ(m[0][1])));
190 
191 
192  if (fabs(p - M_PI_2) < RPY_P_FUZZ) {
193  r = atan2(m[1][0], m[1][1]);
194  p = M_PI_2;
195  /* force it */
196  y = 0.0;
197  } else if (fabs(p + M_PI_2) < RPY_P_FUZZ) {
198  r = -atan2(m[1][2], m[1][1]);
199  p = -M_PI_2;
200  /* force it */
201  y = 0.0;
202  } else {
203  r = atan2(m[1][2], m[2][2]);
204  y = atan2(m[0][1], m[0][0]);
205  }

207 
208  return 0;
