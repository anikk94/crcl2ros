194  int wait ( )
195  {
196  // boost::chrono::high_resolution_clock::time_point interval; /* interval
197  // between this and last wakeup */
198  double numcycles; /* interval, in units of timeout */
199  int missed = 0; /* cycles missed */
200  double remaining = 0.0; /* time remaining until timeout */
201 
202  std::chrono::high_resolution_clock::time_point
203  time_in; /* time wait() was entered */
204  std::chrono::high_resolution_clock::time_point
205  time_done; /* time user function finished */
206 
207  if ( _bSuspend )
208  {
209  std::unique_lock<std::mutex> aLock(_condMutex, std::try_to_lock);
210 
211  // std::mutex::scoped_lock aLock(condMutex);
212  _cWakeup.wait(aLock);
213  }
214 
215  /* first call the user timing function, if any */
216  if ( _function != 0 )
217  {
218  /* set time in */
219  time_in = etime( );
220 
221  if ( ( *_function )(0) == -1 )
222  {
223  return -1; /* fatal error in timing function */
224  }
225  time_done = etime( );
226  }
227  else
228  {
229  /* set time in, time done not used */
230  time_in = etime( );
231  }
232 
233  /* calculate the interval-- for user timing functions, this is how
234  long between this wakeup and the last wakeup. For internal timers,
235  this is how long we need to sleep to make it to the next interval
236  on time. */
237  double interval = ToNanoseconds(time_in - _lastTime) / 1E9;
238  numcycles = interval / _timeout;
239 
240  /* synchronize and set last_time correctly; update idle time */
241  _counts++;
242 
243  if ( _function != 0 )
244  {
245  _lastTime = time_done;
246  }
247  _idle += _timeout > interval ? _timeout - interval : 0;
248  _busy += interval;
249  missed = (int) numcycles;
250  remaining = _timeout - interval;
251  //esleep(remaining);
252  esleepWBreak(remaining);
253  _lastTime = etime( );
254  return missed;
255  }
